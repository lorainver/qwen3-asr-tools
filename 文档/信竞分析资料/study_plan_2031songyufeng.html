1. 题目分析

给定 n 种面值的邮票，每种无限张，求组成 1~m 各面值所需的最少邮票数。经典完全背包/递推问题。

2. 代码分析
现有代码问题：当前 AC 代码实际是素数判断程序，不是邮票问题的解法。需要重写。
3. 参考代码（递推 + DP）
#include<bits/stdc++.h>
using namespace std;
const int MAXN = 100005;
int n, m, stamp[20], dp[MAXN];

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++) cin >> stamp[i];

    // dp[j] = 组成面值 j 所需的最少邮票数
    memset(dp, 0x3f, sizeof(dp));
    dp[0] = 0;

    for (int i = 1; i <= n; i++)           // 每种邮票
        for (int j = stamp[i]; j <= m; j++) // 完全背包正序
            dp[j] = min(dp[j], dp[j - stamp[i]] + 1);

    // 输出 1~m 每个面值的最少邮票数
    for (int j = 1; j <= m; j++)
        cout << dp[j] << (j < m ? ' ' : '\n');

    return 0;
}

B0397 【DP练习】糖果 提高+/省选-

DP背包
5/5 Top用户AC

前5名全部AC | RP加成最高 | 优先级: ★★★

1. 题目分析

有 n 个糖果，每个有重量 a[i]，选择若干个使得总重量恰好为 k 的倍数，最大化选择的糖果数。模 k 背包问题。

2. 代码分析
现有代码分析：使用了 01 背包 + 状态记录。思路正确（dp[j] 记录最大数量，tj[j] 记录实际重量）。
建议：可以简化为只用 dp[j] 表示模 k 余 j 时的最大数量，避免 tj 数组。
3. 优化代码
#include<bits/stdc++.h>
using namespace std;
const int MAXN = 105;
const int MAXK = 1005;
int n, k, a[MAXN], dp[MAXK];

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n >> k;
    for (int i = 1; i <= n; i++) cin >> a[i];

    // dp[j] = 总重量模 k 余 j 时的最大糖果数
    memset(dp, -1, sizeof(dp));
    dp[0] = 0;

    for (int i = 1; i <= n; i++) {
        int tmp[MAXK];
        memcpy(tmp, dp, sizeof(dp));
        for (int j = 0; j < k; j++) {
            if (tmp[j] >= 0) {
                int nj = (j + a[i]) % k;
                dp[nj] = max(dp[nj], tmp[j] + 1);
            }
        }
    }

    cout << dp[0] << endl;
    return 0;
}

B0167 【NOIP2012联赛】开车旅行 普及+/提高

NOIP模拟排序set
5/5 Top用户AC

前5名全部AC | NOIP真题 | 优先级: ★★★

1. 题目分析

NOIP2012 经典题。A 和 B 交替开车，A 每次走最近的未访问城市，B 每次走次近的。给定起始城市和总距离限制，求最远能到达的城市。需要平衡树（set）+ 模拟。

2. 无本地代码，提供模板算法
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int MAXN = 100005;
int n, x0, ans;
ll h[MAXN], dist_a, dist_b;
set<pair<ll,int>> cities;  // (高度, 编号)

// 找到距离 city 最近/次近的城市
int findNearest(int city, int type) {
    // type=1: A走最近, type=2: B走次近
    auto it = cities.lower_bound({h[city], city});
    vector<pair<ll,int>> candidates;
    auto tmp = it;
    for (int i = 0; i < 4 && tmp != cities.end(); i++, tmp++)
        candidates.push_back(*tmp);
    tmp = it;
    if (tmp != cities.begin()) {
        tmp--;
        for (int i = 0; i < 4; i++, tmp--) {
            candidates.push_back(*tmp);
            if (tmp == cities.begin()) break;
        }
    }
    sort(candidates.begin(), candidates.end(),
         [&](const pair<ll,int>&a, const pair<ll,int>&b) {
             ll da = abs(h[a.second] - h[city]);
             ll db = abs(h[b.second] - h[city]);
             return da != db ? da < db : a.second < b.second;
         });
    if (candidates.size() < type) return 0;
    return candidates[type-1].second;
}

int main() {
    cin >> n;
    for (int i = 1; i <= n; i++) { cin >> h[i]; cities.insert({h[i], i}); }
    cin >> x0 >> dist_a >> dist_b;

    int cur = x0, step = 0;
    ll total_a = 0, total_b = 0;
    cities.erase({h[x0], x0});

    while (true) {
        int nxt = findNearest(cur, 1);  // A 走最近
        if (!nxt) break;
        total_a += abs(h[nxt] - h[cur]);
        cities.erase({h[nxt], nxt});
        cur = nxt; step++;

        nxt = findNearest(cur, 2);      // B 走次近
        if (!nxt) break;
        total_b += abs(h[nxt] - h[cur]);
        if (total_a + total_b > dist_a + dist_b) break;
        cities.erase({h[nxt], nxt});
        cur = nxt; step++;
    }
    cout << cur << endl;
    return 0;
}

B0516 【堆练习】哈夫曼编码 普及+/提高

二叉堆哈夫曼树贪心SPJ
5/5 Top用户AC

前5名全部AC | 堆+贪心经典 | 优先级: ★★★

1. 题目分析

给定字符频率，构造哈夫曼编码使得总编码长度最短。经典哈夫曼树 + 最小堆应用。

2. 模板算法（优先队列/最小堆）
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

struct Node {
    ll weight;
    Node *left, *right;
    Node(ll w) : weight(w), left(nullptr), right(nullptr) {}
};

struct cmp {
    bool operator()(Node* a, Node* b) {
        return a->weight > b->weight;
    }
};

int n;
string codes[256];
ll totalBits = 0;

void buildHuffman(priority_queue<Node*, vector<Node*>, cmp>& pq) {
    while (pq.size() > 1) {
        Node* left = pq.top(); pq.pop();
        Node* right = pq.top(); pq.pop();
        Node* parent = new Node(left->weight + right->weight);
        parent->left = left;
        parent->right = right;
        pq.push(parent);
    }
}

void assignCodes(Node* root, string code) {
    if (!root) return;
    if (!root->left && !root->right) {
        codes[(unsigned char)root->weight] = code.empty() ? "0" : code;
        return;
    }
    assignCodes(root->left, code + "0");
    assignCodes(root->right, code + "1");
}

int main() {
    cin >> n;
    priority_queue<Node*, vector<Node*>, cmp> pq;
    for (int i = 0; i < n; i++) {
        char ch; int w;
        cin >> ch >> w;
        pq.push(new Node(w));
    }

    buildHuffman(pq);
    // 计算总编码长度（WPL）
    // assignCodes(pq.top(), "");

    // 输出结果（按题目要求格式）
    Node* root = pq.top();
    ll wpl = 0;
    // 用 BFS 计算 WPL
    queue<pair<Node*,int>> q;
    q.push({root, 0});
    while (!q.empty()) {
        auto [node, depth] = q.front(); q.pop();
        if (!node->left && !node->right) {
            wpl += node->weight * depth;
            continue;
        }
        if (node->left) q.push({node->left, depth+1});
        if (node->right) q.push({node->right, depth+1});
    }
    cout << wpl << endl;
    return 0;
}

B0272 【树练习】FBI树 普及/提高-

树递归NOIP提高
5/5 Top用户AC

前5名全部AC | 优先级: ★★☆

1. 题目分析

给定 01 串，按规则构建 FBI 树：全0为B，全1为I，混合为F，然后递归分治左右子树。NOIP2004真题。

2. 模板算法（递归分治）
#include<bits/stdc++.h>
using namespace std;
int n;
string s;

char buildTree(int l, int r) {
    if (l == r) return s[l] == '0' ? 'B' : 'I';

    // 检查是否全0或全1
    bool all0 = true, all1 = true;
    for (int i = l; i <= r; i++) {
        if (s[i] == '0') all1 = false;
        else all0 = false;
    }

    if (all0) return 'B';
    if (all1) return 'I';

    // 混合 → F，递归构建左右子树
    int mid = (l + r) / 2;
    char left = buildTree(l, mid);
    char right = buildTree(mid + 1, r);
    cout << left << right;  // 后序遍历输出
    return 'F';
}

int main() {
    cin >> n >> s;
    buildTree(0, s.size() - 1);
    cout << endl;
    return 0;
}

输入 01 串 → 检查区间是否全0/全1 → 全0返回B, 全1返回I, 混合返回F并递归左右子树 → 后序遍历输出

1. 题目分析

给定前序和中序遍历，求后序遍历。二叉树基础操作。

2. 模板算法（递归构建）
#include<bits/stdc++.h>
using namespace std;
string pre, in;

void postOrder(int preL, int preR, int inL, int inR) {
    if (preL > preR) return;

    char root = pre[preL];
    int k = inL;
    while (k <= inR && in[k] != root) k++;

    int leftLen = k - inL;
    postOrder(preL + 1, preL + leftLen, inL, k - 1);
    postOrder(preL + leftLen + 1, preR, k + 1, inR);
    cout << root;
}

int main() {
    cin >> pre >> in;
    postOrder(0, pre.size()-1, 0, in.size()-1);
    cout << endl;
    return 0;
}

分治二分数学

1. 题目分析

求解 ax³+bx²+cx+d=0 的实根。可用二分法/牛顿法/分治法。

2. 模板算法（二分法）
#include<bits/stdc++.h>
using namespace std;
double a, b, c, d;

double f(double x) {
    return a*x*x*x + b*x*x + c*x + d;
}

double solve(double l, double r) {
    if (fabs(r - l) < 1e-6) return (l + r) / 2;
    double mid = (l + r) / 2;
    if (f(mid) * f(l) < 0) return solve(l, mid);
    else return solve(mid, r);
}

int main() {
    cin >> a >> b >> c >> d;
    // 在 [-100, 100] 范围内找根
    for (double i = -100; i < 100; i += 1) {
        if (f(i) * f(i+1) <= 0) {
            printf("%.2f ", solve(i, i+1));
        }
    }
    cout << endl;
    return 0;
}
