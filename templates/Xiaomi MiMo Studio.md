const int mod=32768;

struct Matrix {
    long long a[2][2];
    Matrix() { memset(a, 0, sizeof(a)); }
};

Matrix mul(Matrix x, Matrix y) {
    Matrix res;
    for(int i=0; i<2; i++)
        for(int j=0; j<2; j++)
            for(int k=0; k<2; k++)
                res.a[i][j] = (res.a[i][j] + x.a[i][k] * y.a[k][j]) % mod;
    return res;
}

Matrix power(Matrix base, long long exp) {
    Matrix res;
    res.a[0][0] = res.a[1][1] = 1; // 单位矩阵
    while(exp > 0) {
        if(exp & 1) res = mul(res, base);
        base = mul(base, base);
        exp >>= 1;
    }
    return res;
}

int pisano_prime(int p) {
    int upper;
    if (p % 5 == 1 || p % 5 == 4)  // p ≡ ±1 (mod 5)
        upper = p - 1;
    else                             // p ≡ ±2 (mod 5)
        upper = 2 * (p + 1);

    // 枚举 upper 的所有因子，从小到大
    vector<int> factors = get_all_divisors(upper);
    sort(factors.begin(), factors.end());

    // 找最小的因子 k 使得 F(k)≡0, F(k+1)≡1 (mod p)
    for (int k : factors) {
        if (check_period(k, p)) return k;
    }
}
